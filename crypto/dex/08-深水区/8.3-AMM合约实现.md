## 一、最小 AMM 池合约骨架

```solidity
contract MiniPool {
    address public tokenX;
    address public tokenY;
    uint256 public reserveX;   // 池中 X 储备
    uint256 public reserveY;   // 池中 Y 储备
    uint256 public totalLP;    // LP token 总量
    mapping(address => uint256) public lpBalance;

    uint256 constant FEE_BPS = 30;  // 0.3% = 30/10000
}
```

## 二、添加流动性（mint LP）

```solidity
function addLiquidity(uint256 amountX, uint256 amountY) external returns (uint256 lp) {
    _transferFrom(tokenX, msg.sender, address(this), amountX);
    _transferFrom(tokenY, msg.sender, address(this), amountY);

    if (totalLP == 0) {
        lp = sqrt(amountX * amountY);          // 首次: 几何平均(见 8.1)
    } else {
        // 后续: 按比例取小值, 防止有人用错比例占便宜
        lp = min(amountX * totalLP / reserveX,
                 amountY * totalLP / reserveY);
    }
    require(lp > 0, "insufficient liquidity");
    lpBalance[msg.sender] += lp;
    totalLP += lp;
    reserveX += amountX;
    reserveY += amountY;
}
```

## 三、swap（核心：恒定乘积 + 手续费）⭐

```solidity
function swapXforY(uint256 amountXIn, uint256 minYOut) external returns (uint256 yOut) {
    require(amountXIn > 0, "zero in");
    _transferFrom(tokenX, msg.sender, address(this), amountXIn);

    // 扣手续费后参与计算(见 8.1)
    uint256 amountXInWithFee = amountXIn * (10000 - FEE_BPS) / 10000;

    // 恒定乘积: yOut = reserveY * dxEff / (reserveX + dxEff)
    yOut = reserveY * amountXInWithFee / (reserveX + amountXInWithFee);

    require(yOut >= minYOut, "slippage");      // ⭐ 滑点保护! 否则被夹
    require(yOut < reserveY, "insufficient");

    // 更新储备(注意: 加的是含费的全额 amountXIn, 费留在池里给 LP)
    reserveX += amountXIn;
    reserveY -= yOut;

    _transfer(tokenY, msg.sender, yOut);
}
```

> ⭐ `minYOut`（滑点保护）是用户对抗 MEV 夹击的关键（见 [8.4](./8.4-MEV实战剖析.md)）。前端根据预期成交价 + 滑点容忍度算出它。

## 四、移除流动性（burn LP）

```solidity
function removeLiquidity(uint256 lp) external returns (uint256 amountX, uint256 amountY) {
    require(lpBalance[msg.sender] >= lp, "insufficient lp");
    // 按份额取回(含累积的手续费, 因为费已留在 reserve 里)
    amountX = lp * reserveX / totalLP;
    amountY = lp * reserveY / totalLP;

    lpBalance[msg.sender] -= lp;
    totalLP -= lp;
    reserveX -= amountX;
    reserveY -= amountY;

    _transfer(tokenX, msg.sender, amountX);
    _transfer(tokenY, msg.sender, amountY);
}
```

### 3. 整数溢出与精度

```
- 乘除顺序: 先乘后除, 减少精度损失

```

### 4. 防价格操纵 / 闪电贷
```
- 不要把自己的 spot 价格直接暴露给别的协议当预言机(见 3.4)
- 提供 TWAP 累积器(像 Uniswap V2 的 price0CumulativeLast)
```

### 5. 最小流动性锁定
```
- 首次 mint 时永久锁定一小部分 LP(burn 到死地址)
- 防止"首个 LP 通过极小流动性操纵份额价格"的攻击
```

- 滑点保护（minOut）、截止时间（deadline，防交易被长期挂起后以坏价执行）、对 fee-on-transfer/rebasing 代币的兼容性。

## 六、和 CEX 撮合的对照 ⭐

| | [CEX 撮合引擎](../../cex/08-深水区/8.1-撮合引擎实现细节.md) | AMM 合约 |
| --- | --- | --- |
| 核心逻辑 | 订单簿撮合（几百行+定序器+恢复） | 一个乘法恒等式（几十行） |
| 状态 | 内存订单簿 | 链上 reserve |
| 并发 | 单线程定序 | 链的共识天然串行（一个区块内顺序执行） |
| 防错 | 事件溯源/快照 | 链本身保证（不可篡改、可回滚整笔） |
| 防作弊 | 内部风控 | 重入锁/余额校验/滑点保护 |

> 💡 **惊人的对比**：CEX 撮合引擎是个复杂的高性能系统（要自己解决并发、持久化、恢复、高可用），
> 而 AMM 把这一切**外包给了区块链**——合约只需写「一个公式 + 安全校验」，并发/持久化/一致性由链的共识保证。
> **AMM 的极简，本质是把复杂性转移给了底层区块链。** 这是 DeFi「站在区块链肩膀上」的优雅之处（代价是慢和 Gas）。

## 小结

- 最小 AMM 池 = addLiquidity（几何平均 mint）+ swap（恒定乘积+扣费+滑点保护）+ removeLiquidity（按份额取回含费）。
- 滑点保护 minOut 是用户对抗 MEV 的关键。
- 生产级必须处理：重入防护、实际余额校验、溢出精度、防价格操纵(TWAP)、最小流动性锁定、deadline。
- AMM 合约极简（几十行 vs CEX 撮合的复杂系统），因为它把并发/持久化/一致性外包给了区块链共识。
